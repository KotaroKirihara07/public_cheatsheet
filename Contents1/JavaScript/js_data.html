データの取り扱い方法

2024/02/09

このページではデータの取り扱い方法について説明を行う。

JSON

#JSONの文字列をJavaScriptのオブジェクトに展開する

const json_string = '{"id": 2, "name": "aaa"}';
const obj = JSON.parse(json_string);
const id = obj.id;
const name = obj.name;

#JavaScriptのオブジェクトをJSONの文字列に展開する

const obj = {id: 2, name: "aaa"};
const json_string = JSON.stringify(obj);

#JavaScriptのオブジェクトをJSONの文字列に展開する (インデント付き)

const obj = {id: 2, name: "aaa"};
const json_string = JSON.stringify(obj,null,' ');

fetchメソッド

fetchメソッドとは外部ファイルを受信するためのメソッド。

#テキスト(.txt形式ファイル)を読み込む

async function text_load() {
  const data = await fetch('example.txt');
  const text = await data.text();
}

#JSON形式テキストを読み込む

async function json_load() {
  const data = await fetch('example.json');
  const obj = await data.json();
}

#XML形式テキストを読み込む

async function xml_load() {
  const data = await fetch('example.xml');
  const text = await data.text();
  const xml = new DOMParser().parserFromString(text,'application/xml');
}

#バイナリファイルを読み込む

async function binary_load() {
  const data = await fetch('example.jpg');
  const blob = await data.blob();
  const image = new Image();
  image.src = URL.createObjectURL(blob);
}

ローカルデータ

ブラウザ上にデータを保存したり、取り出したりする。

#ローカルストレージへの書き込み

localStorage.setItem('key',data);

#ローカルストレージからの読み出し

const data = localStorage.getItem('key');

#ローカルストレージからキーを一つ削除する

localStorage.removeItem('key_name');

#ローカルストレージをクリアする

localStorage.clear();

#Cookieにデータを保存する

document.cookie = "id = 1";

#Cookieの値をオブジェクトに変換

obj = {};
cookies = document.cookie;
const cookieItems = cookies.split(';');
cookieItems.forEach((item) => {
  const elem = item.split('=');
  const key = elem[0].trim();
  const value = decodeURIComponent(elem[1]);
  obj[key] = value;
});

スマートフォンのセンサ

位置情報

#位置情報を取得する

navigator.geolocation.getCurrentPosition(geoSuccess,geoError);

function geoSuccess(position) {
  \\緯度
  const lat = position.coords.latitude;
  \\経度
  const lng = position.coords.longitude;
  \\緯度経度の誤差
  const ac = position.coords.accuracy;
}

function geoError() {
  alert('Error');
}

ジャイロセンサと加速度センサ

#ジャイロセンサ

function deviceorientationHandler(event) {
  //X軸の傾き
  const x = event.beta;
  //Y軸の傾き
  const y = event.gamma;
  //Z軸の傾き
  const z = event.alpha;
}

#加速度センサ

function devicemotionHandler(event) {
  //X軸の加速度
  const x = event.acceleration.x
  //Y軸の加速度
  const y = event.acceleration.y
  //Z軸の加速度
  const z = event.acceleration.z
}

バイブレーション

#デバイスを振動させる (1000ミリ秒)

navigator.vibrate(1000);

#デバイスを振動、停止、振動させる (1000ミリ秒振動、500ミリ秒停止、1000ミリ秒振動)

navigator.vibrate([1000,500,1000]);
