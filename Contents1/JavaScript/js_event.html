イベントの発生

#設定 (キャプチャフェーズで取得するか、リスナー呼び出しを1回のみにするか、パッシブイベントかどうか)

const option{
  capture: false
  once: true
  passive: false
};

#イベント発生時の関数を設定する (クラス名を指定)

const target = document.querySelector('.class_name');
target.addEventListener(イベント名,関数,option);

#イベント発生時の関数を設定する (id名を指定)

const target = document.querySelector('#id_name');
target.addEventListener(イベント名,関数,option);

#設定したイベントを削除する

target.removeEventListener(イベント名,関数,option);
