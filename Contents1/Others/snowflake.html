実証環境の作成

#データベースの作成
CREATE DATABASE IF NOT EXISTS tech2025_database;

#スキーマの作成
CREATE SCHEMA IF NOT EXISTS tech2025_database.tech2025_schema;

#テーブルの作成
CREATE TABLE IF NOT EXISTS tech2025_database.tech2025_schema.tech2025_table(
  id INT PRIMARY KEY,
  name VARCHAR(100)
  score INT
);

#ウェアハウスの作成
CREATE WAREHOUSE IF NOT EXISTS tech2025_WH
  WAREHOUSE_TYPE = STANDARD
  WAREHOUSE_SIZE = XSMALL
  INITIALLY_SUSPENDED = TRUE;

#ロールの作成
CREATE DATABASE ROLE IF NOT EXISTS tech2025_dbrole;

#ロールに権限を付与
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA tech2025_database.tech2025_schema to ROLE tech2025_dbrole;

#ネットワークポリシーの作成
CREATE NETWORK POLICY IF NOT EXISTS tech2025_nw_policy
  ALLOWED_IP_LIST = ([192.168.100.0/32]);

#PERSON USERの作成
CREATE USER IF NOT EXISTS tech2025_person_user
  PASSWORD = ''
  MUST_CHANGE_PASSWORD = FALSE
  DISABLED = FALSE
  DEFAULT_WAREHOUSE = tech2025_WH
  DEFAULT_NAMESPACE = tech2025_database.tech2025_schema
  DEFAULT_ROLE = tech2025_dbrole
  TYPE = PERSON;
  NETWORK_POLICY = tech2025_nw_policy;

#PERSON USERにロールを付与する
GRANT ROLE tech2025_dbrole TO USER tech2025_person_user;

#SERVICE USERの作成
CREATE USER IF NOT EXISTS tech2025_service_user
  DISABLED = FALSE
  DEFAULT_WAREHOUSE = tech2025_WH
  DEFAULT_NAMESPACE = tech2025_database.tech2025_schema
  DEFAULT_ROLE = tech2025_dbrole
  RSA_PUBLIC_KEY =
  RSA_PUBLIC_KEY_FP =
  TYPE = SERVICE;
  NETWORK_POLICY = tech2025_nw_policy;

#SERVICE USERにロールを付与する
GRANT ROLE tech2025_dbrole TO USER tech2025_service_user;

#データの登録
INSERT INTO tech2025_table VALUES
  (1,'a-001',10),
  (2,'a-002',20),
  (3,'a-003',30),
  (4,'a-004',40),
  (5,'a-005',50),
  (6,'a-006',60),
  (7,'a-007',70),
  (8,'a-008',80),
  (9,'a-009',90),
  (10,'a-010',100);

実証環境の削除

#ネットワークポリシーの削除
DROP NETWORK POLICY IF EXISTS tech2025_nw_policy;

#ロールの削除
DROP DATABASE ROLE IF EXISTS tech2025_dbrole;

#ウェアハウスの削除
DROP WAREHOUSE IF EXISTS tech2025_WH;

#テーブルの削除
DROP TABLE IF EXISTS tech2025_database.tech2025_schema.tech2025_table CASCADE;

#スキーマの削除
DROP SCHEMA IF EXISTS tech2025_schema CASCADE;

#データベースの削除
DROP DATABASE IF EXISTS tech2025_database CASCADE;
