Docker Engineのインストール

#UbuntuにおけるDocker Engineのインストール方法

  1. sudo apt-get install apt-transport-https ca-certificates curl gnupg-agent software-properties-common
  2. curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
  3. sudo apt-key fingerprint 0EBFCD88
  4. sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"
  5. sudo apt-get update
  6. sudo apt-get install docker-ce docker-ce-cli containerd.io
  7. sudo usermod -aG docker $USER
  8. docker --version

#Amazon LinuxにおけるDocker Engineのインストール方法

  1. sudo yum update -y
  2. sudo yum install docker
  3. sudo service docker start
  4. sudo usermod -a -G docker ec2-user
  5. docker --version

Docker Composeのインストール

#Amazon LinuxにおけるDocker Composeのインストール方法 ( https://github.com/docker/compose/releases )

  1. sudo mkdir -p /usr/local/lib/docker/cli-plugins
  2. sudo curl -SL https://github.com/docker/compose/releases/download/v2.21.0/docker-compose-linux-x86_64 -o /usr/local/lib/docker/cli-plugins/docker-compose
  3. sudo chmod +x /usr/local/lib/docker/cli-plugins/docker-compose
  4. docker compose version
