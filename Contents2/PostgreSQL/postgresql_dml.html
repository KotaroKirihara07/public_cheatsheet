DML (Data Manipulation Language)

2024/08/17

DML (Data Manipulation Language)はデータ操作言語と呼ばれ、テーブルの行を検索したり、変更したりする。

SELECT

#特定の列を取り出す

SELECT col_name1, col_name2 FROM Table_name;

#すべての列を取り出す

SELECT * FROM Table_name;

#列に別名をつける(日本語の場合は""で囲む)

SELECT col_name1 AS aaa, col_name2 AS bbb FROM Table_name;

#列から重複要素を省いた形式で取り出す

SELECT DISTINCT col_name FROM Table_name;

#条件を満たす行を取り出す

SELECT col_name1, col_name2 FROM Table_name WHERE col_name1 = 'aaa';

#計算(3列目に1列目の1.08倍の列を作成)

SELECT col_name1, col_name2, col_name1 * 1.08 AS col_name3 FROM Table_name;

#演算子

演算子 意味
= 等しい
<> 等しくない
>= 以上
> より大きい
<= 以下
< より小さい

#NULLである行を選択する (NULLに対しては上記演算子が使用不可)

SELECT col_name1, col_name2 FROM Table_name WHERE col_name2 IS NULL;

#NULLでない行を選択する (NULLに対しては上記演算子が使用不可)

SELECT col_name1, col_name2 FROM Table_name WHERE col_name2 IS NOT NULL;

#NOT演算子

SELECT col_name1, col_name2 FROM Table_name WHERE NOT col_name1 > 10;

#AND演算子

SELECT col_name1, col_name2 FROM Table_name WHERE col_name1 = 10 AND col_name2 = 20;

#OR演算子

SELECT col_name1, col_name2 FROM Table_name WHERE col_name1 = 10 OR col_name2 = 20;

#AND演算子とOR演算子を組み合わせる

SELECT col_name1, col_name2, col_name3 FROM Table_name WHERE col1_name1 > 10 AND (col_name2 = 10 OR col_name3 = 20);

#行ロックをかける

SELECT * FROM Table_name WHERE col1_name1 > 10 FOR UPDATE;

#集約関数

集約関数 意味 例
COUNT テーブルの行数を数える SELECT COUNT(col_name1) FROM Table_name;
SUM テーブルの数値列のデータを合計する SELECT SUM(col_name1) FROM Table_name;
AVG テーブルの数値列のデータの平均を計算する SELECT AVG(col_name1) FROM Table_name;
MAX テーブルの数値列のデータの最大値 SELECT MAX(col_name1) FROM Table_name;
MIN テーブルの数値列のデータの最小値 SELECT MIN(col_name1) FROM Table_name;
※COUNTでは、列名を*することでテーブルの行数が得られ、通常の列名を引数とするとNULLの行を除外した行数が得られる。
※SUM, AVG, MAX, MINでは SELECT SUM(col_name1), SUM(col_name2) FROM Table_name;のように複数の計算を同時に行える。

#重複値を除外した行数を数える

SELECT COUNT(DISTINCT col_name1) FROM Table_name;

GROUP BY句

テーブルを集約する。SELECT句で選択できる列は「定数」、「集約関数」、「GROUP BY句で指定した列」のみである。

#集約した後、行数を数える

SELECT col_name1, COUNT(*) FROM Table_name GROUP BY col_name1;

HAVING句

集約した結果に対する条件指定。HAVING句で選択できる列は「定数」、「集約関数」、「GROUP BY句で指定した列」のみである。

#集約した後、行数を数え,行数が4の行のみ取り出す

SELECT col_name1, COUNT(*) FROM Table_name GROUP BY col_name1 HAVING COUNT(*) = 4;

ORDER BY句

検索結果を並び替える。NULLは先頭または末尾に表示される。

#col_name1を基準にして並び替える (昇順)

SELECT col_name1, col_name2 FROM Table_name ORDER BY col_name1 ASC;

#col_name1を基準にして並び替える (降順)

SELECT col_name1, col_name2 FROM Table_name ORDER BY col_name1 DESC;

#col_name1を第一基準、col_name2を第二基準にして並び替える (昇順)

SELECT col_name1, col_name2, col_name3 FROM Table_name ORDER BY col_name1 ASC, col_name2 ASC;

LIMIT句

特定の行数のレコードのみ取得する。

#先頭の10行のみ取得する

SELECT * FROM Table_name LIMIT 10;

#5行目から10行目のみ取得する

SELECT * FROM Table_name LIMIT 4,10;

#5行目から10行目のみ取得する (OFFSETの使用)

SELECT * FROM Table_name LIMIT 4 OFFSET 10;

サブクエリ

使い捨てのViewであり、使用後は消去される。

スカラサブクエリ : 戻り値が単一の値(平均値や最小値や最大値など)であるサブクエリである。定数や列名を記述することができるすべての場所に配置することができる。

相関サブクエリ : 小分けしたグループ内での比較をするためのもの。結合条件を必ずサブクエリ内に配置しなければならない。


#サブクエリ

SELECT col_name1, col_name2
  FROM (SELECT ----- )
  AS New_table_name;

#スカラ・サブクエリ

SELECT col_name1, col_name2
  FROM Table_name1
  WHERE col_name2 IN
  (SELECT col_name2 FROM Table_name2);

#相関サブクエリ

SELECT col_name1, col_name2
  FROM Table_name1
  WHERE EXISTS
  (SELECT * FROM Table_name2 WHERE Table_name1.col_name1 = Table_name2.col_name1);

INSERT

データを登録する。

INSERT INTO Table_name (col_name1, col_name2, col_name3) VALUES (value1, value2, value3);

#列名を省略して登録する

INSERT INTO Table_name VALUES (value1, value2, value3);

#NULLを登録する

INSERT INTO Table_name (col_name1, col_name2, col_name3) VALUES (value1, value2, NULL);

#明示的にデフォルト値を登録する

INSERT INTO Table_name (col_name1, col_name2, col_name3) VALUES (value1, value2, DEFAULT);

#暗黙的にデフォルト値を登録する (col_name2をデフォルト値にする)

INSERT INTO Table_name (col_name1, col_name3) VALUES (value1, value3);

#他のテーブルにコピーする

INSERT INTO New_table_name (col_name1,col_name2) SELECT col_name1,col_name2 FROM Old_table_name;

UPDATE

#登録済みのデータを変更する。

UPDATE Table_name SET col_name = 200;

#特定の行だけ変更する

UPDATE Table_name SET col_name1 = 200 WHERE col_name2 = 'note';

#特定の行だけNULLに変更する

UPDATE Table_name SET col_name1 = NULL WHERE col_name2 = 'note';

#特定の複数列を変更する

UPDATE Table_name SET col_name1 = 200, col_name2 = 'www' WHERE col_name3 = 'pen';

DELETE

#テーブルを保持したままデータのみすべて削除する。

DELETE FROM Table_name;

#条件を満たす行だけ削除する

DELETE FROM Table_name WHERE col_name1 >= 200;

TRUNCATE

DELETE文より処理速度が速いが、条件によるレコードの削除ができない。

#テーブルを全行削除する。

TRUNCATE Table_name;
