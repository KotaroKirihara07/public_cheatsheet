forms.py

#forms.ModelFormによるフォームの作成 

from django import forms
form .models import model_name
                
class form_name(forms.ModelForm):
    class Meta:
        model = model_name
        fields = ['name', 'age', ... ]

#forms.Formによるフォームの作成 

from django import forms

class form_name(forms.Form):
    name = forms.CharField(label='name')
    age = forms.IntegerField(label='age')

入力フィールド

入力フィールド クラス 説明
forms.CharField(label='ラベル',required=True,min_length=min,max_length=max) CharField <input type="text">というタグを生成
forms.EmailField(label='ラベル',required=True,min_length=min,max_length=max) EmailField <input type="email">というタグを生成
forms.IntegerField(label='ラベル',required=True,min_value=min,max_value=max) IntegerField <input type="number">というタグを生成
forms.FloatField(label='ラベル',required=True,min_value=min,max_value=max) FloatField <input type="number">というタグを生成
forms.URLField(label='ラベル',required=True,min_length=min,max_length=max) URLField <input type="url">というタグを生成
form.DateField(label='DateField',input_formats=['%d']) DateField
form.TimeField(label='TimeField',input_formats=['%H','%M','%S']) TimeField
form.DateTimeField(label='DateTimeField',input_formats=['%y','%m','%d','%H','%M','%S']) DateTimeField
forms.BooleanField(label='ラベル',required=False) BooleanField
forms.NullBooleanField(label='ラベル',required=False) NullBooleanField
forms.ChoiceField(label='ラベル',choices=data) ChoiceField チョイス
forms.ChoiceField(label='ラベル',choices=data,widget=forms.RadioSelect()) ChoiceField ラジオボタン
forms.ChoiceField(label='ラベル',choices=data,widget=forms.Select()) ChoiceField 単一選択リスト
forms.MultiChoiceField(label='ラベル',choices=data,widget=forms.SelectMultiple()) MultiChoiceField 複数選択リスト

#ChoiceFieldにおけるchoices(=data)の設定方法

data = [
    # (値, ラベル)
    ("1", "one"),
    ("2", "two"),
    ("3", "three")
]

バリデーション


バリデーション (forms.ModelForm)

forms.ModelFormにおけるバリデーションは「models.py」で設定する。


バリデーション (forms.Form)

#バリデーションのチェック 

if (form.is_valid()):
    #formにエラーがない場合の処理
else:
    #formにエラーがあった場合の処理

#CharField/EmailField/URLFieldのバリデーション

  • required : 必須項目か否か(True or False)
  • min_length : 最小文字数(整数値)
  • max_length : 最大文字数(整数値)
  • empty_value : 空の入力を許可するか否か(True or False)

#IntegerField/FloatFieldのバリデーション

  • required : 必須項目か否か(True or False)
  • min_value : 最小値(整数値)
  • max_value : 最大値(整数値)

#DateField/TimeField/DateTimeFieldのバリデーション

  • input_formats=[フォーマット1, フォーマット2, ...] : フォーマット形式を指定する

#DateField/TimeField/DateTimeFieldのフォーマット形式

  • '%y' : 年
  • '%m' : 月
  • '%d' : 日
  • '%H' : 時
  • '%M' : 分
  • '%S' : 秒

#エラーを発生させる

raise form.ValidationError('エラーメッセージ')
