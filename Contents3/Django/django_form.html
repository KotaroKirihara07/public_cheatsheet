forms.Form

#フォームの作成

from django import forms

class form_name(forms.Form):
    name = forms.CharField(label='name')
    age = forms.IntegerField(label='age')

バリデーション (forms.ModelForm)

forms.ModelFormにおけるバリデーションは「models.py」で設定する。


バリデーション (forms.Form)

#バリデーションのチェック 

if (form.is_valid()):
    #formにエラーがない場合の処理
else:
    #formにエラーがあった場合の処理

#CharField/EmailField/URLFieldのバリデーション

  • required : 必須項目か否か(True or False)
  • min_length : 最小文字数(整数値)
  • max_length : 最大文字数(整数値)
  • empty_value : 空の入力を許可するか否か(True or False)

#IntegerField/FloatFieldのバリデーション

  • required : 必須項目か否か(True or False)
  • min_value : 最小値(整数値)
  • max_value : 最大値(整数値)

#DateField/TimeField/DateTimeFieldのバリデーション

  • input_formats=[フォーマット1, フォーマット2, ...] : フォーマット形式を指定する

#DateField/TimeField/DateTimeFieldのフォーマット形式

  • '%y' : 年
  • '%m' : 月
  • '%d' : 日
  • '%H' : 時
  • '%M' : 分
  • '%S' : 秒

#エラーを発生させる

raise form.ValidationError('エラーメッセージ')
