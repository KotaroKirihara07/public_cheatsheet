クラス・関数

2024/07/13

このページではクラスや関数についての説明を行う。

クラス

#クラスの定義

class Person():
  #変数の定義
  __slots__ = ['a','b','c']

  #コンストラクタ
  def __init__(self):
    self.a = 0
    self.b = 1

  #関数
  def func1(self,num):
    self.c = num
    print(self.c)

  #クラスメソッド
  @classmethod
  def func2(cls,value):
    return value*2

  #スタティックメソッド
  @staticmethod
  def func3(value):
    return value*2

#インスタンス化

Bob = Person(4)
Bob.func1("Bob")

#staticメソッドの実行 (インスタンス化しなくても使用可能)

Person.func2(100)

#継承

class SalesPerson(Person):
  def __init__(self,var):
    super().__init__(var):
      #処理
  def func3(self):
    #基底クラスの関数の呼び出す
    super.func2(2)

関数

#関数の定義

def func(a,b):
  c = a + b
  return c

#グローバル変数

global var

#ラムダ式 (引数なし)

func = lambda : print("出力")
func()

#ラムダ式 (引数あり)

func = lambda n : n**2
func(2)

#ラムダ式 (if文つき)

func = lambda n,m : "Good" if n*m > 10 else "Bad"
func(1,2)
