このページではクラスや関数についての説明を行う。

クラス

#クラスの定義

class Person():
    #クラスがもつことができるインスタンス変数の固定
    __slots__ = ['a','b','c']

    #クラス変数の定義
    FIXED_VALUE = 100

    #コンストラクタ
    def __init__(self):
        self.a = 0
        self.b = 1

    #インスタンスメソッドの定義
    def func1(self,num):
        self.c = num
        print(self.c)

    #クラスメソッドの定義 (クラス(cls)にアクセスできる and インスタンス化しなくても使用可能なメソッド)
    @classmethod
    def func2(cls,value):
        return cls.FIXED_VALUE * value

    #スタティックメソッド (インスタンス化しなくても使用可能なメソッド) の定義
    @staticmethod
    def func3(value):
        return value*2

    #pythonコマンドで実行されたときの処理
    if __name__ == '__main__':
        #処理

#インスタンス化 and メソッドの実行

Bob = Person(4)
Bob.func1("Bob")

#スタティックメソッドの実行 (インスタンス化しなくても使用可能)

Person.func2(100)

#継承

class SalesPerson(Person):
    def __init__(self,var):
        super().__init__(var):
            #処理
    def func3(self):
        #基底クラスの関数の呼び出す
        super.func2(2)
