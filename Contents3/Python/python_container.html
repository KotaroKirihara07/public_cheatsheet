リスト

#リストの生成

list1 = [1,2,3]
list2 = [val*2 for val in list1]
list3 = [val*2 for val in list1 if val % 2 == 1]
list4 = [i for i in range(0,10)]

#スライス (nからmまで)

list1[n:m+1]

#指定したインデックスの要素を削除

del list1[index]

#ソートして新しいリストを返す

list2 = sorted(list1)

#逆順のリストを返す

rev_list = list1[::-1]

#リスト内の要素を加工して新しいリストを作成する

list2 = map(lambda v: v**2, list1)

#リスト内の要素を特定の条件で絞り込む

list2 = filter(lambda v: v < 5, list1)

#リスト内の要素を順に処理して1つにまとめる

list2 = reduce(lambda result, v : result*v , list1)

#メソッド集

メソッド 説明 戻り値の型
list1.append(var) 末尾に変数を追加 void
list1.insert(N, var) N番目に変数を追加 void
list1.remove(var) 指定した要素を削除 void
list1.pop(idx) 指定したインデックスの要素を削除して返す (引数なしの場合は末尾の要素を返す) var
list1.index(var) 引数で指定した値が格納されているインデックスを返す int
list1.sort() ソートして元のリストを変更する void
list1.extend(list2) リスト1にリスト2を結合させる void
",".join(list1) リストをcsv文字列に変換する str
list1.clear() リストを削除する void
list1.count(var) リストにvarが何回登場するか int
list1.reverse() リストの内容を逆順にする void

#タプルの生成

t = (1,2,3)

t[2]

t[start:stop:step]

x in t

集合

#集合の生成

s = {1,2,3,4}
s = {val*2 for val in list1}
s = {val*2 for val in list1 if val % 2 == 1}

#リストから集合の生成

s = set(list1)

#要素が含まれているか(True or False)

4 in s

#メソッド集

メソッド 説明 戻り値の型
s.add(5) 要素の追加 void
s.remove(5) 指定した要素の削除 void
s.pop() 末尾の要素を返して削除 var
s1.union(s2) 和集合 set
s1.intersection(s2) 積集合 set
s1.difference(s2) 差集合 set
s1.issubset(s2) s1がs2に含まれているか boolean
s1.issuperset(s2) s2がs1に含まれているか boolean

辞書

#dictionaryの生成

dic = {1:10,2:20,3:30}
dic[4] = 40
dic2 = {val: 0 for val in list1}
dic3 = {val: 0 for val in list1 if val % 2 == 0}

#listからdictionaryの生成

dic = dict(zip(key_list, value_list))

#指定したキーに対応する値を取得

dic[2]

#指定したキーの要素を削除する

del dic[3]

#存在の判定(True or False)

1 in dic.keys()
10 in dic.values()
(1,10) in dic.items()

#valueが最大のkeyの取得

max_val_key = max(dic,key=dic.get)

#valueが最小のkeyの取得

min_val_key = min(dic,key=dic.get)

#辞書をキーでソートしたリストを返す

sorted_bykey_list = sorted(dic.items(), key=lambda x: (x[0]))

#辞書を値でソートしたリストを返す

sorted_byvalue_list = sorted(dic.items(), key=lambda x: (x[1]))

#メソッド集

メソッド 説明 戻り値の型
dic.setdefault(key, value) 指定したキーに値がなければ与える void
dic1.update(dic2) 2つの辞書をマージする void
dic.get(key, default) 指定したキーに対応する値を返す(デフォルトの設定も可能) var
dic.keys() すべてのキーをリスト化 list
dic.values() すべての値をリスト化 list
dic.items() すべてのキーと値の組をタプル化 tuple
dic.copy() 辞書のコピー(ハードコピー) dict
dic.pop(key) 指定したキーの要素を削除して返す var
dic.clear() すべて削除 void
