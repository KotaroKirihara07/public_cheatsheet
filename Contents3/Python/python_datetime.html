datetime

2025/02/11

このページではdatetimeモジュールについての説明を行う。

モジュールの読み込み

#モジュールの読み込み

import datetime

datetime型

#2020年1月1日1時10分10秒100ミリ秒のdatetime型を生成

d = datetime.datetime(2020,1,1,1,10,10,100)

#現在のdatetime型を生成

d = datetime.datetime.now()

#現在のdatetime型を生成 (UTC)

d = datetime.datetime.utcnow()

#属性集

属性 説明 戻り値の型
d.year 年 int
d.month 月 int
d.day 日 int
d.hour 時間 int
d.minute 分 int
d.second 秒 int
d.milisecond マイクロ秒 int

#メソッド集

メソッド 説明 戻り値の型
d.date() 日付部分を取得 datetime.date
d.time() 時刻部分を取得 datetime.time
d.timetz() 時刻部分を取得 datetime.time
d.timestamp() タイムスタンプ値を取得 datetime.timestamp
d.toordinal() 西暦1年1月1日からの通算日を取得する int
d.weekday() 曜日を取得する (0:月曜日, 6:日曜日) int
d.isoweekday() 曜日を取得する (1:月曜日, 7:日曜日) int
d.isocalendar() (ISO年, ISO週番号, ISO曜日)のタプルを取得 tuple

#文字列からdatetime型への変換

d = datetime.strptime("日付文字列", "フォーマット文字列")

#datetimeから文字列への変換

d.strftime("フォーマット文字列")

#フォーマット文字列

フォーマット文字列 説明
"%Y" 西暦4桁
"%m" 2桁でゼロ埋めした月
"%d" 2桁でゼロ埋めした日
"%H" 2桁でゼロ埋めした時(24時間表記)
"%M" 2桁でゼロ埋めした分
"%S" 2桁でゼロ埋めした秒
"%f" 6桁でゼロ埋めしたマイクロ秒

#フォーマット文字列の例1

"%Y/%m/%d %H:%M:%S:%f"

#フォーマット文字列の例2

"%Y-%m-%d %H:%M:%S:%f"

date型

#2020年1月1日のdate型を生成

d = datetime.date(2020,1,1)

#現在のdateを取得する

d = datetime.date.today()

time型

#time型の生成

t = datetime.time(hour=0, minute=0, second=0, microseconds=0)

#13時40分30秒のtime型の生成

t = datetime.time(13,40,30)

timedelta型

#timedelta型の生成

timedelta = datetime.timedelta(days=0, seconds=0, microseconds=0, milliseconds=0, minutes=0, hours=0, weeks=0)

#10日後の日付を計算する

d2 = d + datetime.timedelta(days=10)

timezone型

#日本時間

tz = datetime.timezone(datetime.timedelta(hours=+9), 'Asia/Tokyo')
