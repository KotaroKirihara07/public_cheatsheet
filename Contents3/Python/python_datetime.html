datetime

2024/02/11

このページではdatetimeモジュールについての説明を行う。

モジュールの読み込み

#モジュールの読み込み

from datetime import datetime, date, time, timedelta

datetime型

#2020年1月1日1時10分10秒100ミリ秒のdatetime型を生成

d = datetime(2020,1,1,1,10,10,100)

#メソッド集

メソッド 説明 戻り値の型
d.year 年 int
d.month 月 int
d.day 日 int
d.hour 時間 int
d.minute 分 int
d.second 秒 int
d.milisecond マイクロ秒 int

#文字列からdatetime型への変換

datetime.strptime("日付文字列", "フォーマット文字列")

#datetimeから文字列への変換

d.strftime("フォーマット文字列")

#フォーマット文字列

フォーマット文字列 説明
"%Y" 西暦4桁
"%m" 2桁でゼロ埋めした月
"%d" 2桁でゼロ埋めした日
"%H" 2桁でゼロ埋めした時(24時間表記)
"%M" 2桁でゼロ埋めした分
"%S" 2桁でゼロ埋めした秒
"%f" 6桁でゼロ埋めしたマイクロ秒

#フォーマット文字列の例1

"%Y/%m/%d %H:%M:%S:%f"

#フォーマット文字列の例2

"%Y-%m-%d %H:%M:%S:%f"

#現在のdatetimeを取得する

dt = datetime.now()

date型

#2020年1月1日のdate型を生成

d = date(2020,1,1)

#現在のdateを取得する

dt = date.today()

time型

#time型の生成

t = time(hour=0, minute=0, second=0, microseconds=0)

#13時40分30秒のtime型の生成

T = time(13,40,30)

timedelta型

#timedelta型の生成

timedelta = timedelta(days=0, seconds=0, microseconds=0, milliseconds=0, minutes=0, hours=0, weeks=0)

#10日後の日付を計算する

dt2 = d + timedelta(days=10)
