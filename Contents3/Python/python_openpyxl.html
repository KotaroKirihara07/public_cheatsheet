openpyxl

2023/06/09

openpyxlとは、ExcelファイルをPythonで操作するためのライブラリ。

ライブラリの読み込み

#ライブラリの読み込み

import openpyxl

ワークブックの操作

#ワークブックを開く

wb = openpyxl.load_workbook("file_name.xlsx")

#ワークブックの保存(名前を変更することで別名新規保存も可能)

wb.save("file_name.xlsx")

#ワークブックを閉じる

wb.close()

シートの操作

#シートの取得 (上から、シート名、インデックス、アクティブなシート)

ws = wb["Sheet1"]
ws2 = wb.worksheets[0]
ws3 = wb.active

#シートの作成

ws4 = wb.create_sheet(title="Sheet2")

#シートの削除

wb.remove(ws4)

#シートリストの取得

ws_list = wb.worksheets

セルの操作

#セルの取得

cell1 = ws["B4"]
cell2 = ws.cell(row=4, column=2)
cell3 = ws.cell(4, 2)

#セルの値の取得

val1 = ws["B4"].value
val2 = ws.cell(row=4, column=2).value
val3 = ws.cell(4, 2).value
val4 = cell1.value

#セルの値の書き込み

ws["B4"] = "www"
ws.cell(row=4, column=2, value= 4)

#セルの値の削除

ws["B4"] = " "
ws.cell(row=4, column=2, value=" ")
