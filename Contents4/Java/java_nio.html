java.nioパッケージ

2024/05/05

fileパッケージ ： FileSystemsクラス、Filesクラス、Pathsクラス、StandardCopyOptionクラス

FileSystemsクラス

ファイル・システム内のファイルやその他のオブジェクトにアクセスするためのクラス。

#import宣言

import java.nio.file.FileSystems;

#FileSystemの取得

var fs = FileSystems.getDefault();

#ディレクトリをPathオブジェクトとして受け取る

var dir = fs.getPath("C:/Data");

#ファイルをPathオブジェクトとして受け取る

var path = fs.getPath("C:/Data/sample.txt");

Filesクラス

ファイル/ディレクトリを操作するためのクラス。

#import宣言

import java.nio.file.Files;

#ファイルを操作するためのメソッド集

メソッド 説明 戻り値の型
Files.exists(path) ファイルが存在するか boolean
Files.isReadable(path) ファイルが読み取り可能か boolean
Files.isWritable(path) ファイルが書き込み可能か boolean
Files.isExecutable(path) ファイルが実行可能か boolean
Files.size(path) ファイルのサイズ int
Files.copy(path1,fs.getPath("C:/data/sample.txt"),StandardCopyOption.REPLACE_EXISTING) ファイルをコピー (存在する場合は置換) var
Files.move(path1,fs.getPath("C:/data/sample.txt"),StandardCopyOption.REPLACE_EXISTING) ファイルを移動 (存在する場合は置換) void
Files.move(path1,fs.getPath("C:/data/filename.txt"),StandardCopyOption.REPLACE_EXISTING) ファイル名を変更 var
Files.delete(path) ファイルを削除 void
Files.deleteIfExists(path) ファイルが存在するときのみファイルを削除 void

#ディレクトリを操作するためのメソッド集

メソッド 説明 戻り値の型
Files.createDirectories(dir) ディレクトリの作成 void
Files.exists(dir) ディレクトリが存在するか boolean
Files.isDirectory(dir) ディレクトリかどうか boolean
Files.copy(dir1,fs.getPath("C:/data/sample"),StandardCopyOption.REPLACE_EXISTING) ディレクトリをコピー var
Files.move(dir1,fs.getPath("C:/data/box"),StandardCopyOption.REPLACE_EXISTING) ディレクトリを移動 void
Files.delete(fs.getPath("C:/data/box")) ディレクトリを削除 void
Files.deleteIfExists(fs.getPath("C:/data/box")) ディレクトリが存在するときのみディレクトリを削除 void
Files.list(dir) dir内のサブディレクトリ/ファイルの一覧をStreamで出力する var

#ファイルへの書きこみ

//ファイルを開く
var writer = Files.newBufferedWriter(path,StandardOpenOption.APPEND);
//書き込み
writer.write("書き込み内容");
//改行
writer.newLine();
//ファイルを閉じる
writer.close();

#ファイルの読み込み

//ファイルを開く
var reader = Files.newBufferedReader(path);
//読み込み
while ((line = reader.readline()) != null) {
  System.out.println(line);
}
//ファイルを閉じる
reader.close();

#オープンモード

設定値 説明
StandardOpenOption.CREATE ファイルが存在しない場合に新規作成
StandardOpenOption.APPEND 追記モードでファイルを開く
StandardOpenOption.CREATE_NEW 新規にファイルを作成(存在する場合はエラー)
StandardOpenOption.DELETE_ON_CLOSE 閉じるときにファイルを削除する

Pathsクラス

pathを扱うためのクラス。

#pathの指定

var path = Paths.get("C:/Data/sample.txt")

StandardCopyOptionクラス

説明。

#import宣言

import java.nio.file.StandardCopyOption;

ファイルの読み込み方法 (一行ごとに文字列として格納)

import java.nio.file.Files;
import java.nio.file.FileSystems;
import java.io.IOException;

public class Readfile {
    public static void main(String[] args) {
        var fs = FileSystems.getDefault();
        var path = fs.getPath("C:\\---\\Book1.csv");

        try (var reader = Files.newBufferedReader(path)) {
            var line = "";
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
            reader.close();
        }
        catch (IOException e) {
            e.printStackTrace();
        }
    }
}

ファイルの読み込み方法 (リストに格納)

import java.nio.file.Files;
import java.nio.file.FileSystems;
import java.io.IOException;
import java.util.ArrayList;

public class Readfile {
    public static void main(String[] args) {
        var fs = FileSystems.getDefault();
        var path = fs.getPath("C:\\---\\Book1.csv");

        //それぞれの項目を保存するためのリストを作成する
        var id_list = new ArrayList<Integer>();
        var name_list = new ArrayList<String>();
        var price_list = new ArrayList<Integer>();

        try (var reader = Files.newBufferedReader(path)) {
            //項目名をリストから除くためのフラグ
            boolean Flag = false;

            var line = "";
            while ((line = reader.readLine()) != null) {
                String[] csv = line.split(",");

                // ファイルの二行目からリストに格納する
                if (Flag) {
                    id_list.add(Integer.parseInt(csv[0]));
                    name_list.add(csv[1]);
                    price_list.add(Integer.parseInt(csv[2]));
                }
                Flag = true;
            }
            reader.close();
        }
        catch (IOException e) {
            e.printStackTrace();
        }
    }
}

ファイルへの書き込み方法 (文字列を末尾に追加する)

ファイルへの書き込み方法 (リストをcsv形式で保存)

import java.util.Arrays;
import java.util.ArrayList;
import java.nio.file.Files;
import java.nio.file.StandardOpenOption;
import java.nio.file.FileSystems;
import java.io.IOException;

public class Writefile {
    public static void main(String[] args) {
        var fs = FileSystems.getDefault();
        var path = fs.getPath("C:\\---\\Book1.csv");

        var id_list = new ArrayList<Integer>(Arrays.asList(11,12,13,14,15));
        var name_list = new ArrayList<String>(Arrays.asList("aa","bb","cc","dd","ee"));
        var price_list = new ArrayList<Integer>(Arrays.asList(1000,2000,3000,4000,5000));

        try (var writer = Files.newBufferedWriter(path,StandardOpenOption.APPEND)) {
            for (int i = 0; i < id_list.size(); i++) {
                //リストから値を取り出す
                String id = String.valueOf(id_list.get(i));
                String name = String.valueOf(name_list.get(i));
                String price = String.valueOf(price_list.get(i));
                String data = id + "," + name + "," + price;

                //書き込み
                writer.write(data);
                //改行
                writer.newLine();
            }
            //ファイルを閉じる
            writer.close();
        }
        catch (IOException e) {
            e.printStackTrace();
        }
    }
}
