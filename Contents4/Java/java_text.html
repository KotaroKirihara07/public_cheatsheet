NumberFormatクラス 、 SimpleDateFormatクラス

NumberFormatクラス

数値を成形するためのクラス。

#import宣言

import java.text.NumberFormat;

#汎用数値フォーマット

var nf = NumberFormat.getNumberInstance();
var fnum = nf.format(num);

#通貨フォーマット

var nf = NumberFormat.getCurrencyInstance(Locale.JAPAN);
var fnum = nf.format(num);

#整数フォーマット

var nf = NumberFormat.getIntegerInstance();
var fnum = nf.format(num);

#パーセントフォーマット

var nf = NumberFormat.getPercentInstance();
var fnum = nf.format(num);

#フォーマット設定のためのメソッド

メソッド 説明 戻り値の型
nf.setCurrency(Currency currency) 通貨 void
nf.setGroupingUsed(boolean tf) 桁区切り文字(,)を使用するか void
nf.setMaximumFractionDigits(int max) 小数部分の最大桁数 void
nf.setMaximumIntegerDigits(int max) 整数部分の最大桁数 void
nf.setMinimumFractionDigits(int min) 小数部分の最小桁数 void
nf.setMinimumIntegerDigits(int min) 整数部分の最大桁数 void

SimpleDateFormatクラス

#import宣言

import java.text.SimpleDateFormat;

#インスタンス化

SimpleDateFormat sdf = new SimpleDateFormat(”フォーマットパターン”)

#フォーマットパターン

フォーマットパターン 出力例
yyyy-MM-dd 2020-01-01
yyyy年 MM月 dd日 2020年 01月 01日
yyyy-MM-dd-HH-mm-ss.SSS 2020-01-01-13-30-40.509

#タイムゾーンを設定

setTimeZone

#指定されたDateを日付文字列にフォーマットする

Date date = new Date();
String str = sdf.format(date);

#文字列から解析してDateオブジェクトを生成する

Date date = sdf.parse(str);
