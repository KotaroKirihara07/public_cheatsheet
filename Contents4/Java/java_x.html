javaxパッケージとはオープンで標準的なAPIを実装したオプションパッケージである。

javax.namingパッケージ : InitialContextクラス

javax.sqlパッケージ : DataSourceクラス

InitialContextクラスとDataSourceクラス

データベースに接続するためのクラス。

#import宣言

import javax.naming.InitialContext;
import javax.sql.DataSource;
import java.sql.Connection;

#データベースへの接続

Context context = new InitialContext();
DataSource ds = (DataSource)context.lookup("java:comp/env/jdbc/database_name");
try (Connection con = ds.getConnection()) {
    //SQLの処理 (詳しくは java.sqlパッケージを参照 )
} catch (SQLException e) {
   e.printStackTrace();
}
