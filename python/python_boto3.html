#バケット一覧を取得

s3 = boto3.client('s3')

#バケット内のオブジェクトの一覧を取得

s3 = boto3.resource('s3')
bucket = s3.Bucket('BUCKET_NAME')
object_summary_iterator = bucket.objects.all()
object_list = [object.key for object in object_summary_iterator]
