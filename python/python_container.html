リスト

#リストの生成

list1 = [1,2,3]
list2 = [val*2 for val in list1]
list3 = [val*2 for val in list1 if val % 2 == 1]
list4 = [i for i in range(0,10)]

#スライス (nからmまで)

list1[n:m+1]

#指定したインデックスの要素を削除

del list1[index]

#ソートして新しいリストを返す

list2 = sorted(list1)

#逆順のリストを返す

rev_list = list1[::-1]

#関数集

関数 説明 戻り値の型
all(list1) リスト内のすべての要素がTrueか bool
any(list1) リスト内の要素に少なくとも1つTrueがあるか bool
map(lambda v: v**2, list1) リスト内の要素を加工してリストを返す list
filter(lambda v: v < 5, list1) 特定の条件を満たす要素のみのリストを返す list
reduce(lambda result, v : result*v , list1) リスト内の要素を順に処理して1つにまとめる list

#メソッド集

メソッド 説明 戻り値の型
list1.append(x) 末尾に変数を追加 -
list1.insert(N, x) N番目に変数を追加 -
list1.remove(x) 指定した要素を削除 -
list1.pop(idx) 指定したインデックスの要素を削除して返す (引数なしの場合は末尾の要素を返す) -
list1.index(x) 引数で指定した値が格納されているインデックスを返す int
list1.sort() ソートして元のリストを変更する -
list1.extend(list2) リスト1にリスト2を結合させる -
",".join(list1) リストをcsv文字列に変換する str
list1.clear() リストを削除する -
list1.count(x) リストにxが何回登場するか int
list1.reverse() リストの内容を逆順にする -

#集合の生成

s = {1,2,3,4}

#リストから集合の生成

s = set(list1)

#メソッド集

メソッド 説明 戻り値の型
s.add(x) 要素の追加 -
s.remove(x) 指定した要素の削除 -
s.pop() 末尾の要素を返して削除 var
s1.union(s2) 和集合 set
s1.intersection(s2) 積集合 set
s1.difference(s2) 差集合 set
s1.issubset(s2) s1がs2に含まれているか bool
s1.issuperset(s2) s2がs1に含まれているか bool

辞書

#dictionaryの生成

dic1 = {1:10,2:20,3:30}
dic1[4] = 40
dic2 = {val: 0 for val in list1}
dic3 = {val: 0 for val in list1 if val % 2 == 0}

#listからdictionaryの生成

dic1 = dict(zip(key_list, value_list))

#指定したキーに対応する値を取得

dic1[2]

#指定したキーの要素を削除する

del dic1[3]

#存在の判定(True or False)

1 in dic1.keys()
10 in dic1.values()
(1,10) in dic1.items()

#valueが最大のkeyの取得

max_val_key = max(dic1, key=dic1.get)

#valueが最小のkeyの取得

min_val_key = min(dic1, key=dic1.get)

#辞書をキーでソートしたリストを返す

sorted_bykey_list = sorted(dic1.items(), key=lambda x: (x[0]))

#辞書を値でソートしたリストを返す

sorted_byvalue_list = sorted(dic1.items(), key=lambda x: (x[1]))

#メソッド集

メソッド 説明 戻り値の型
dic1.setdefault(key, value) 指定したキーに値がなければ与える -
dic1.update(dic2) 2つの辞書をマージする -
dic1.get(key, default) 指定したキーに対応する値を返す (デフォルトの設定も可能) var
dic1.keys() すべてのキーをリスト化 list
dic1.values() すべての値をリスト化 list
dic1.items() すべてのキーと値の組をタプル化 tuple
dic1.copy() 辞書のコピー(シャローコピー) dict
dic1.pop(key) 指定したキーの要素を削除して返す var
dic1.clear() すべて削除 -
