#ライブラリの読み込み

import matplotlib.animation as animation

ArtistAnimation

#gif形式で保存する方法

  1. fig = plt.figure()
  2. 空のリストimgsを作成する
  3. im = plt.plot(x,y)として画像を作成する
  4. imgs.append(im)として画像をリストに保存する
  5. ani = animation.ArtistAnimation(fig, imgs, interval=100)
    6. ※1秒間に(1000/interval)枚の画像を表示させる
  6. ani.save('anime_sample.gif', writer='pillow') として保存

#mp4形式で保存する方法

  1. ffmpegをダウンロードする
  2. plt.rcParams["animation.ffmpeg_path"] = "C:\\ffmpeg/bin/ffmpeg.exe"
  3. fig = plt.figure()
  4. 空のリストimgsを作成する
  5. im = plt.plot(x,y)として画像を作成する
  6. imgs.append(im)として画像をリストに保存する
  7. ani = animation.ArtistAnimation(fig, imgs, interval=100)
    8. ※1秒間に(1000/interval)枚の画像を表示させる
  8. ani.save('anime_sample.mp4') として保存
